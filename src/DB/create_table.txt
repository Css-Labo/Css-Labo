

//ユーザテーブル
CREATE TABLE user_tbl(
    user_id     INT         AUTO_INCREMENT,
    user_name   VARCHAR(30) NOT NULL    ,
    user_pass   VARCHAR(30) NOT NULL    ,
    user_mail   VARCHAR(30) NOT NULL    ,
    PRIMARY KEY (user_id)
);

//CSSテーブル
CREATE TABLE css_tbl(
    css_id      INT         AUTO_INCREMENT,
    css_name    VARCHAR(20) NOT NULL    ,
    css_code    VARCHAR(100)NOT NULL    ,
    css_tag     VARCHAR(100)            ,
    css_like    INT         default 0   ,
    PRIMARY KEY (css_id)
);

//お気に入りテーブル
CREATE TABLE bookmark_tbl(
    user_id     NOT NULL    REFERENCES  user_tbl(user_id),
    css_id      NOT NULL    REFERENCES  css_tbl(css_id)
    PRIMARY KEY (user_id)
    PRIMARY KEY (css_id)
);

//コメントテーブル
CREATE TABLE comment_tbl(
    thread_id   INT         AUTO_INCREMENT,
    css_id      NOT NULL    REFERENCES  css_tbl(css_id)
    PRIMARY KEY (thread_id)
);
